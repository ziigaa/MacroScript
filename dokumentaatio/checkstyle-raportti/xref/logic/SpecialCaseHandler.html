48      /**
49       * Interprets an if-sentence input from the script by splitting it and
50       * processing the remaining data.
51       *
52       * @param ifSentence The sentence to interpret
53       * @return If the requirements of the if/sentence are met, then the command
54       * after the condition is returned. Otherwise an empty string is returned.
55       */
56      public String handleIf(String ifSentence) {
57  
58          String toReturn = "";
59  
60          try {
61              String ifSplittedBySpaces[] = ifSentence.split(" ");
62              String splitByEqual[] = ifSplittedBySpaces[1].split("=");
63  
64              int givenValue = 0;
65  
66              try {
67                  givenValue = Integer.parseInt(splitByEqual[1]);
68              } catch (NumberFormatException e) {
69                  if (variablesAndValues.containsKey(splitByEqual[1])) {
70                      givenValue = variablesAndValues.get(splitByEqual[1]);
71                  }
72              }
73  
74              if (variablesAndValues.containsKey(splitByEqual[0])) {
75                  if (variablesAndValues.get(splitByEqual[0]) == givenValue) {
76                      for (int i = 2; i < ifSplittedBySpaces.length; i++) {
77                          toReturn = toReturn + ifSplittedBySpaces[i] + " ";
78                      }
79                      toReturn = toReturn.substring(0, toReturn.length() - 1);
80                  }
81              } else {
82                  myLogger.insert("Variable not found in: " + ifSentence);
83                  toReturn = "3rror";
84                  //variable not found
85              }
86          } catch (ArrayIndexOutOfBoundsException e) {
87              //ei "="-merkkiä
88              String ifSplittedBySpaces[] = ifSentence.split(" ");
89              if (ifSplittedBySpaces[1].contains("findColor")) {
90                  //System.out.println("löyty findColor");
91                  String colorPoint = handleFindColorAsParameter(ifSentence);
92                  if (!colorPoint.contains("-1")) {
93                      try {
94                          //System.out.println("ifS: " + ifSentence);
95  
96                          int beginCommand = 7;
97  
98                          if (ifSplittedBySpaces[1].contains("FromPalette")) {
99                              beginCommand--;
100                         }
101 
102                         if (ifSentence.contains("StartingFromPoint")) {
103                             beginCommand += 2;
104                         }
105 
106                         for (int i = beginCommand; i < ifSplittedBySpaces.length; i++) {
107                             toReturn = toReturn + ifSplittedBySpaces[i] + " ";
108                         }
109                         toReturn = toReturn.substring(0, toReturn.length() - 1);
110                     } catch (Exception ex) {
111                         myLogger.insert("Invalid format in: " + ifSentence + "\nIf-sentence needs to have a condition");
112                         //if-lauseella ei ollut ehtoa
113                     }
114                 } else {
115                     myLogger.insert("Color not found: " + ifSentence);
116                     //väriä ei löytynyt
117                     toReturn = "3rror";
118                 }
119             } else {
120                 myLogger.insert("Invalid format in: " + ifSentence);
121             }
122         } catch (NullPointerException e) {
123             //ei ollu parametrii
124             myLogger.insert("Invalid parameter format in: " + ifSentence);
125             toReturn = "3rror";
126         }
127         return toReturn;
128     }

130     /**
131      * Handles cases where a variable is given a value, either by assining a
132      * simple integer value to it or by a calculation. t. ex. i=3 or i=i*5
133      *
134      * @param commandLine The line of script-code that does not start with a
135      * command
136      */
137     public void handleVariableSettingAndVariableCalculations(String commandLine) {
138 
139         try { //joko muuttujalle ollaan asettamassa arvoa, tai sitten komento on tuntematon
140 
141             String splitByEqual[];
142             splitByEqual = commandLine.split("=");
143 
144             //System.out.println("splByEQ0=" + splitByEqual[0]);
145             if (variablesAndValues.containsKey(splitByEqual[0])) {
146                 try {
147                     variablesAndValues.put(splitByEqual[0], Integer.parseInt(splitByEqual[1]));
148                 } catch (NumberFormatException e) {
149 
150                     if (variablesAndValues.containsKey(splitByEqual[1])) {
151                         variablesAndValues.put(splitByEqual[0], variablesAndValues.get(splitByEqual[1]));
152                         return;
153                     }
154 
155                     String splitByOperation[];
156                     int operationType = 0;
157 
158                     if (splitByEqual[1].contains("+")) {
159                         splitByOperation = splitByEqual[1].split("\\+");
160                         operationType = 1;
161                     } else if (splitByEqual[1].contains("-")) {
162                         splitByOperation = splitByEqual[1].split("-");
163                         operationType = 2;
164                     } else if (splitByEqual[1].contains("/")) {
165                         splitByOperation = splitByEqual[1].split("/");
166                         operationType = 3;
167                     } else if (splitByEqual[1].contains("*")) {
168                         splitByOperation = splitByEqual[1].split("\\*");
169                         operationType = 4;
170                     } else {
171                         //syntax error
172                         myLogger.insert("Syntax error1: " + commandLine);
173                         splitByOperation = "010".split("1");
174                         return;
175                     }
176 
177                     int value1 = 0, value2 = 0;
178 
179                     try {
180                         value1 = Integer.parseInt(splitByOperation[0]);
181                     } catch (NumberFormatException ex) {
182                         if (variablesAndValues.containsKey(splitByOperation[0])) {
183                             value1 = variablesAndValues.get(splitByOperation[0]);
184                         } else {
185                             myLogger.insert("Syntax error2: " + commandLine);
186                             //syntax error
187                         }
188                     }
189 
190                     try {
191                         value2 = Integer.parseInt(splitByOperation[1]);
192                     } catch (NumberFormatException ex) {
193                         if (variablesAndValues.containsKey(splitByOperation[1])) {
194                             value2 = variablesAndValues.get(splitByOperation[1]);
195                         } else {
196                             myLogger.insert("Syntax error3: " + commandLine);
197                             //syntax error
198                         }
199                     }
200 
201                     int result = 0;
202 
203                     switch (operationType) {
204                         case 0:
205                             myLogger.insert("Syntax error, unknown operation: " + commandLine);
206                             //error
207                             break;
208                         case 1:
209                             result = value1 + value2;
210                             break;
211                         case 2:
212                             result = value1 - value2;
213                             break;
214                         case 3:
215                             result = (int) Math.round((double) value1 / value2);
216                             break;
217                         case 4:
218                             result = value1 * value2;
219                             break;
220                     }
221 
222                     variablesAndValues.put(splitByEqual[0], result);
223 
224                 } catch (NullPointerException e) {
225                     myLogger.insert("Syntax error, parameters seem to be missing: " + commandLine);
226                     //ei ollu mitää O.O
227                 }
228 
229             } else {
230                 myLogger.insert("Unknown command or variable does not exist: " + commandLine);
231                 //erröööööör, ei oo tehty muuttujaa
232             }
233         } catch (Exception e) {
234             myLogger.insert("Syntax error, unknown command: " + commandLine);
235             //errööööör, ei voi splitata, joten tuntematon komento.
236         }
237     }

239     /**
240      * Adds a variable in the HashMap that holds all script variables and their
241      * values.
242      *
243      * @param rawData Either a name for the variable in which case it will be
244      * assigned the default value of 0, or a variable name and a value for it.
245      * t. ex. i=3.
246      */
247     public void createVariable(String rawData) {
248 
249         if (!rawData.contains("=") && !rawData.contains(" ")) {
250             myLogger.insert("No value was set for " + rawData + ", so default (=0) will be used.");
251             variablesAndValues.put(rawData, 0); //asetetaan oletus 0.
252             return;
253         }
254 
255         try {
256             String splitByEqual[] = rawData.split("=");
257             if (variablesAndValues.containsKey(splitByEqual[0])) {
258                 //parametri on jo olemassa
259                 myLogger.insert("A variable called " + splitByEqual[0] + " already exists. Specifications: " + rawData + " were not set.");
260             } else {
261                 try {
262                     variablesAndValues.put(splitByEqual[0], Integer.parseInt(splitByEqual[1]));
263                 } catch (ArrayIndexOutOfBoundsException e) { //ei arvoa
264                     myLogger.insert("No value was set for " + splitByEqual[0] + ". Skipping line.");
265                 }
266             }
267         } catch (NumberFormatException e) {
268             String splitByEqual[] = rawData.split("=");
269             if (variablesAndValues.containsKey(splitByEqual[1])) {
270                 variablesAndValues.put(splitByEqual[0], variablesAndValues.get(splitByEqual[1]));
271             } else {
272                 myLogger.insert("Syntax error: " + splitByEqual[0] + " needs to get its value as a number or from another variable.");
273             }
274             //ei oo numero
275         } catch (NullPointerException e) {
276             myLogger.insert("Syntax error: No value was set for the variable");
277             //ei ollu parametrii
278         }
279     }

281     /**
282      * When any color finding operation is used from the class ColorOperator it
283      * needs to be processed here.
284      *
285      * @param commandLine The command line containing the color finding
286      * operation
287      * @return The coordinates as String. Format "x y" t. ex. "50 100"
288      */
289     public String handleFindColorAsParameter(String commandLine) {
290         String splitBySpaces[] = commandLine.split(" ");
291         int method = 0;
292         boolean rightColorFormat = false;
293 
294         if (splitBySpaces[1].equals("findColor")) {
295             method = 1;
296         } else if (splitBySpaces[1].equals("findColorStartingFromPoint")) {
297             method = 2;
298         } else if (splitBySpaces[1].equals("findColorFromPalette")) {
299             method = 3;
300         } else if (splitBySpaces[1].equals("findColorFromPaletteStartingFromPoint")) {
301             method = 4;
302         } else {
303             myLogger.insert("Syntax error5: " + commandLine);
304             //tuntematon findColor
305             return "3rror";
306         }
307 
308         int parameterPlaceFix = 1;
309 
310         if (method == 1 || method == 2) {
311             if (splitBySpaces[2].length() == 7 && splitBySpaces[2].charAt(0) == '#' && isValidHex(splitBySpaces[2]) == true) {
312                 parameterPlaceFix = 1;
313                 rightColorFormat = true;
314             }
315         } else if (method == 3 || method == 4) {
316             if (!colorPalette.isEmpty()) {
317                 parameterPlaceFix = 0;
318                 rightColorFormat = true;
319             }
320         }
321 
322         if (rightColorFormat == false) {
323             if (method == 1 || method == 2) {
324                 myLogger.insert("The color input is in a faulty format: " + commandLine);
325             } else if (method == 3 || method == 4) {
326                 myLogger.insert("ColorPalette is empty: " + commandLine);
327             }
328             return "3rror"; //joko väri ei ollut hex-muodossa tai jos käytössä on colorPalette, niin se oli tyhjä
329         }
330 
331         Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
332         int screenWidth = (int) screenSize.getWidth();
333         int screenHeight = (int) screenSize.getHeight();
334         Point startPoint = new Point(-1, -1);
335         int x1 = 0, x2 = 0, y1 = 0, y2 = 0, startX = 0, startY = 0;
336 
337         try {
338 
339             x1 = Integer.parseInt(splitBySpaces[2 + parameterPlaceFix]);
340             y1 = Integer.parseInt(splitBySpaces[3 + parameterPlaceFix]);
341             x2 = Integer.parseInt(splitBySpaces[4 + parameterPlaceFix]);
342             y2 = Integer.parseInt(splitBySpaces[5 + parameterPlaceFix]);
343 
344             if (method == 2 || method == 4) {
345                 startX = Integer.parseInt(splitBySpaces[6 + parameterPlaceFix]);
346                 startY = Integer.parseInt(splitBySpaces[7 + parameterPlaceFix]);
347                 if (isBetween(startX, x1, x2) == false || isBetween(startY, y1, y2) == false) {
348                     myLogger.insert("The start point needs to be in the area: " + commandLine);
349                     return "3rror"; //aloituspiste ei alueella
350                 }
351                 startPoint.x = startX;
352                 startPoint.y = startY;
353             }
354 
355         } catch (Exception e) {
356             myLogger.insert("Syntax error6: " + commandLine);
357             return "3rror"; //epäkelpoja numeroita tai parametrejä puuttuu
358         }
359 
360         if (isBetween(x1, 0, screenWidth) == false
361                 || isBetween(y1, 0, screenHeight) == false
362                 || isBetween(x2, x1, screenWidth) == false
363                 || isBetween(y2, y1, screenHeight) == false) {
364             myLogger.insert("The points need to be placed within the correct pixel range: " + commandLine + "\nThe correct range for x is from 0 to " + screenWidth + " and for y from 0 to " + screenHeight);
365             return "3rror"; //pisteet eivät ole näytön alueella
366         }
367 
368         if (x1 == x2 || y1 == y2) {
369 
370             myLogger.insert("Rectangle width and height must be > 0: " + commandLine);
371             return "3rror";
372         }
373 
374         Rectangle area = new Rectangle();
375         area.setRect(x1, y1, x2 - x1, y2 - y1);
376         Point colorCoodrinates = new Point(-1, -1);
377 
378         switch (method) {
379             case 1:
380                 colorCoodrinates = colorOperation.findColor(colorOperation.hexToRGB(splitBySpaces[2]), area);
381                 break;
382             case 2:
383                 colorCoodrinates = colorOperation.findColorStartingFromPoint(colorOperation.hexToRGB(splitBySpaces[2]), startPoint, area);
384                 break;
385             case 3:
386                 colorCoodrinates = colorOperation.findColorFromPalette(colorPalette, area);
387                 break;
388             case 4:
389                 colorCoodrinates = colorOperation.findColorFromPaletteStartingFromPoint(colorPalette, startPoint, area);
390                 break;
391         }
392 
393         return colorCoodrinates.x + " " + colorCoodrinates.y;
394     }

396     /**
397      * This is used to check if a given integer value is inbetween the given
398      * parameters.
399      *
400      * @param value The value which is compared to the parameters
401      * @param low Parameter for the lowest possible number (inclusive)
402      * @param high Parameter for the highest possible number (inclusive)
403      * @return true if the value is between the given parameters. False if the
404      * number is not between the given parameters.
405      */
406     private boolean isBetween(int value, int low, int high) {
407         if (value <= high && value >= low) {
408             return true;
409         } else {
410             return false;
411         }
412     }

414     /**
415      * Checks if the given parameter is a valid number in hex format. For
416      * example #FFFFFF and #000000 are valid.
417      *
418      * @param hex The String to check
419      * @return True if String is a hex number, false if not.
420      */
421     public boolean isValidHex(String hex) {
422         for (int i = 1; i < hex.length(); i++) {
423             if (Character.digit(hex.charAt(i), 16) == -1) {
424                 return false;
425             }
426         }
427         return true;
428     }
