12  import java.awt.AWTException;
13  import java.awt.Robot;
14  import static java.awt.event.KeyEvent.*;
15  import java.util.Random;
16  import java.util.logging.Level;
17  import java.util.logging.Logger;
18  
19  public class KeyboardOperator {
20  
21      private Robot bot;
22      private boolean isHuman = false;
23      private boolean keyFunctionDown = false;
24      private boolean keyFunctionUp = false;
25  
26      public KeyboardOperator() throws AWTException {
27          this.bot = new Robot();
28          this.isHuman = false;
29      }
30  
31      public void keyDown(char theKey) {
32          this.keyFunctionDown = true;
33          typeChar(theKey, this.isHuman, this.keyFunctionDown, this.keyFunctionUp);
34      }
35  
36      public void keyUp(char theKey) {
37          keyFunctionUp = true;
38          typeChar(theKey, this.isHuman, this.keyFunctionDown, this.keyFunctionUp);
39      }
40  
41      public void pressEnter(boolean isHuman) {
42          if (isHuman == true) {
43              doRndSleep();
44          }
45          bot.keyPress(VK_ENTER);
46          if (isHuman == true) {
47              doRndSleep();
48          }
49          bot.keyRelease(VK_ENTER);
50      }
51  
52      public void type(CharSequence characters, boolean isHuman) {
53          int length = characters.length();
54          this.isHuman = isHuman;
55          for (int i = 0; i < length; i++) {
56              char character = characters.charAt(i);
57              typeChar(character, isHuman, keyFunctionDown, keyFunctionUp);
58          }
59      }
60  
61      public void typeChar(char charToType, boolean isHuman, boolean onlyDown, boolean onlyUp) {
62          String alphas = "abcdefghijklmnopqrstuvwxyz";
63          String numbers = "0123456789";
64          String specialCharSet1 = "!\"#Â¤%&/()=";
65          String specialCharSet2 = ",.-";
66          int charInt = 0;
67          boolean shiftNeeded = false;
68  
69          if (alphas.contains(Character.toString(charToType))) {
70              charInt = alphas.indexOf(charToType) + VK_A;
71          } else if (alphas.toUpperCase().contains(Character.toString(charToType))) {
72              charInt = alphas.toUpperCase().indexOf(charToType) + VK_A;
73              shiftNeeded = true;
74          } else if (numbers.contains(Character.toString(charToType))) {
75              charInt = numbers.indexOf(charToType) + VK_0;
76          } else if (charToType == ' ') {
77              charInt = VK_SPACE;
78          } else if (specialCharSet1.contains(Character.toString(charToType))) {
79              shiftNeeded = true;
80  
81              charInt = VK_0 + 1 + specialCharSet1.indexOf(charToType);
82  
83              if (charToType == '=') {
84                  charInt = VK_0;
85              }
86          } else if (specialCharSet2.contains(Character.toString(charToType))) {
87              if (charToType == '.') {
88                  charInt = VK_PERIOD;
89              } else if (charToType == ',') {
90                  charInt = VK_COMMA;
91              } else if (charToType == '-') {
92                  charInt = VK_MINUS;
93              }
94          }
95          if (isHuman == true) {
96              doRndSleep();
97          }
98          if (shiftNeeded == true) {
99              bot.keyPress(VK_SHIFT);
100             if (isHuman == true) {
101                 doRndSleep();
102             }
103         }
104         if (onlyDown == true) {
105             onlyDown = false;
106             bot.keyPress(charInt);
107             return;
108         }
109         if (onlyUp == true) {
110             onlyUp = false;
111             bot.keyRelease(charInt);
112             return;
113         }
114 
115         bot.keyPress(charInt);
116         if (isHuman == true) {
117             doRndSleep();
118         }
119 
120         bot.keyRelease(charInt);
121 
122         if (shiftNeeded == true) {
123             if (isHuman == true) {
124                 doRndSleep();
125             }
126             bot.keyRelease(VK_SHIFT);
127             shiftNeeded = false;
128         }
129     }

131     public void doRndSleep() {
132         Random rnd = new Random();
133         long sInterval = (long) rnd.nextInt(300);
134 
135         try {
136             Thread.sleep(sInterval);
137         } catch (InterruptedException ex) {
138             Logger.getLogger(MouseOperator.class.getName()).log(Level.SEVERE, null, ex);
139         }
140     }
