11  import java.awt.PointerInfo;
12  import java.util.Timer;

14  import java.util.logging.Level;
15  import java.util.logging.Logger;

18   *

21  public class MacroScript {
22  
23      /**
24       * @param args the command line arguments
25       */
26      public static void main(String[] args) {
27          // TODO code application logic here
28          frmMain mainWindow = new frmMain();
29          mainWindow.setVisible(true);
30  
31          Timer tmrCursorPosition = new Timer();
32  
33          tmrCursorPosition.scheduleAtFixedRate(
34                  new TimerTask() {
35              public void run() {
36                  //while (true) {
37                  PointerInfo pointInfo = MouseInfo.getPointerInfo();
38                  Point pntNow = pointInfo.getLocation();
39                  int x = (int) pntNow.getX();
40                  int y = (int) pntNow.getY();
41  
42                  mainWindow.lblXpos.setText(String.valueOf(x));
43                  mainWindow.lblYpos.setText(String.valueOf(y));
44  
45                  if (mainWindow.getColorPickerActive() == true) {
46                      try {
47                          mainWindow.setColors();
48                      } catch (AWTException ex) {
49                          Logger.getLogger(MacroScript.class.getName()).log(Level.SEVERE, null, ex);
50                      }
51                  }
52              }
53          },
54                  0, // run first occurrence immediately
55                  100);  // run every 100ms
56      }

58  }
