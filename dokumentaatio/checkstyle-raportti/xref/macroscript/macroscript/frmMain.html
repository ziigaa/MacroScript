343     /**
344      * Captures the color from current mouse coordinates (retrieved from lblXpos
345      * and lblYpos.
346      *
347      * @return The color capture as a Color object
348      * @throws AWTException If unable to create a new Java Robot
349      */
350     public Color getColorUnderCursor() throws AWTException {
351         Robot robot = new Robot();
352         Color color = robot.getPixelColor(Integer.parseInt(lblXpos.getText()), Integer.parseInt(lblYpos.getText()));
353         return color;
354     }

356     /**
357      * Captures the color from current mouse coordinates (retrieved from lblXpos
358      * and lblYpos. Then converts the retrieved Color object to String while
359      * changing the format to hex code.
360      *
361      * @return The hex string of the captured color t. ex. #000000 or #FFFFFF
362      * @throws AWTException If unable to create a new Java Robot
363      */
364     public String getColorHexUnderCursor() throws AWTException {
365         Robot robot = new Robot();
366         Color color = robot.getPixelColor(Integer.parseInt(lblXpos.getText()), Integer.parseInt(lblYpos.getText()));
367         String hexValue = "#" + toHexString(color.getRed()) + toHexString(color.getGreen()) + toHexString(color.getBlue());
368         return hexValue.toUpperCase();
369     }

415     private String toHexString(final int input) {
416         String out = Integer.toHexString(input);
417         if (out.length() == 1) {
418             out = "0" + out;
419         }
420         return out;
421     }

423     /**
424      * Opens a file from the prompted location and displays its contents in
425      * txtScript after asking the user if he wants to save the current script.
426      *
427      * To increase efficiency this method uses a BufferedReader for buffering a
428      * line of script at a time instead of one character.
429      */
430     public void loadFile() {
431 
432         String txtData = txtScript.getText().trim();
433 
434         if (!txtData.equals("")) {
435             int dialogButton = JOptionPane.YES_NO_OPTION;
436             int dialogResult = JOptionPane.showConfirmDialog(null, "Would you like to save your current script first?", "Save first?", dialogButton);
437             if (dialogResult == JOptionPane.YES_OPTION) {
438                 saveAs();
439             }
440         }
441 
442         txtScript.setText("");
443 
444         FileNameExtensionFilter extensionFilter = new FileNameExtensionFilter("Great MacroScript file", "gms");
445         final JFileChooser loadFileChooser = new JFileChooser();
446         loadFileChooser.setApproveButtonText("Open");
447         loadFileChooser.setFileFilter(extensionFilter);
448         int actionDialog = loadFileChooser.showOpenDialog(this);
449         if (actionDialog != JFileChooser.APPROVE_OPTION) {
450             return;
451         }
452 
453         // !! File fileName = new File(loadFileChooser.getSelectedFile() + ".gms");
454         File file = loadFileChooser.getSelectedFile();
455 
456         BufferedReader inFile = null;
457         boolean beginning = true;
458         try {
459             inFile = new BufferedReader(new FileReader(file));
460             String str;
461             while ((str = inFile.readLine()) != null) {
462                 if (beginning == false) {
463                     txtScript.append("\n");
464                 }
465                 txtScript.append(str);
466                 beginning = false;
467             }
468         } catch (IOException e) {
469         } finally {
470             try {
471                 inFile.close();
472             } catch (Exception ex) {
473             }
474         }
475     }

477     /**
478      * Saves the text from txtScript into a file according to the users input.
479      * Uses a BufferedWriter, because its efficiency in repetitive file saving.
480      */
481     public void saveAs() {
482         FileNameExtensionFilter extensionFilter = new FileNameExtensionFilter("Great MacroScript file", "gms");
483         final JFileChooser saveAsFileChooser = new JFileChooser();
484         saveAsFileChooser.setApproveButtonText("Save");
485         saveAsFileChooser.setFileFilter(extensionFilter);
486         int actionDialog = saveAsFileChooser.showOpenDialog(this);
487         if (actionDialog != JFileChooser.APPROVE_OPTION) {
488             return;
489         }
490 
491         // !! File fileName = new File(SaveAs.getSelectedFile() + ".gms");
492         File file = saveAsFileChooser.getSelectedFile();
493         if (!file.getName().endsWith(".gms")) {
494             file = new File(file.getAbsolutePath() + ".gms");
495         }
496 
497         BufferedWriter outFile = null;
498         try {
499             outFile = new BufferedWriter(new FileWriter(file));
500 
501             txtScript.write(outFile);
502 
503         } catch (IOException ex) {
504             ex.printStackTrace();
505         } finally {
506             if (outFile != null) {
507                 try {
508                     outFile.close();
509                 } catch (IOException e) {
510                 }
511             }
512         }
