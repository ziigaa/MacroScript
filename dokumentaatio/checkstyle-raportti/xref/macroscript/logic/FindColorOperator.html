6   package macroscript.logic;

8   import macroscript.gui.frmLogger;
9   import java.awt.Dimension;
10  import java.awt.Point;
11  import java.awt.Rectangle;
12  import java.awt.Toolkit;
13  import java.util.ArrayList;
14  import java.util.HashMap;
15  import macroscript.actions.ColorOperator;
16  import macroscript.actions.ColorOperator;

18  /**
19   * For handling color finding operations within the script.
20   *
21   * @author Joonas
22   */
23  public class FindColorOperator {
24  
25      HashMap<String, Integer> variablesAndValues;
26      ArrayList<String> colorPalette;
27      private final ColorOperator colorOperation;
28      private final frmLogger myLogger;
29  
30      /**
31       * The constructor.
32       *
33       * @param variablesAndValues A HashMap containing all the variables and
34       * their values.
35       * @param colorPalette The ArrayList for colorPalette variable within the
36       * script.
37       * @param myLogger The error logger.
38       */
39      public FindColorOperator(HashMap<String, Integer> variablesAndValues, ArrayList<String> colorPalette, frmLogger myLogger) {
40          this.variablesAndValues = new HashMap<>();
41          this.variablesAndValues = variablesAndValues;
42          this.colorOperation = new ColorOperator();
43          this.colorPalette = new ArrayList<>();
44          this.colorPalette = colorPalette;
45          this.myLogger = myLogger;
46      }
47  
48      /**
49       * When any color finding operation is used from the class ColorOperator it
50       * needs to be processed here.
51       *
52       * @param commandLine The command line containing the color finding
53       * operation
54       * @return The coordinates as String. Format "x y" t. ex. "50 100"
55       */
56      public String handleFindColorAsParameter(String commandLine) {
57          String splitBySpaces[] = commandLine.split(" ");
58          int method = 0;
59          boolean rightColorFormat = false;
60  
61          if (splitBySpaces[1].equals("findColor")) {
62              method = 1;
63          } else if (splitBySpaces[1].equals("findColorStartingFromPoint")) {
64              method = 2;
65          } else if (splitBySpaces[1].equals("findColorFromPalette")) {
66              method = 3;
67          } else if (splitBySpaces[1].equals("findColorFromPaletteStartingFromPoint")) {
68              method = 4;
69          } else {
70              myLogger.insert("Syntax error5: " + commandLine);
71              //tuntematon findColor
72              return "3rror";
73          }
74  
75          int parameterPlaceFix = 1;
76  
77          if (method == 1 || method == 2) {
78              if (splitBySpaces[2].length() == 7 && splitBySpaces[2].charAt(0) == '#' && isValidHex(splitBySpaces[2]) == true) {
79                  parameterPlaceFix = 1;
80                  rightColorFormat = true;
81              }
82          } else if (method == 3 || method == 4) {
83              if (!colorPalette.isEmpty()) {
84                  parameterPlaceFix = 0;
85                  rightColorFormat = true;
86              }
87          }
88  
89          if (rightColorFormat == false) {
90              if (method == 1 || method == 2) {
91                  myLogger.insert("The color input is in a faulty format: " + commandLine);
92              } else if (method == 3 || method == 4) {
93                  myLogger.insert("ColorPalette is empty: " + commandLine);
94              }
95              return "3rror"; //joko väri ei ollut hex-muodossa tai jos käytössä on colorPalette, niin se oli tyhjä
96          }
97  
98          Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
99          int screenWidth = (int) screenSize.getWidth();
100         int screenHeight = (int) screenSize.getHeight();
101         Point startPoint = new Point(-1, -1);
102         int x1 = 0, x2 = 0, y1 = 0, y2 = 0, startX = 0, startY = 0;
103 
104         try {
105 
106             x1 = Integer.parseInt(splitBySpaces[2 + parameterPlaceFix]);
107             y1 = Integer.parseInt(splitBySpaces[3 + parameterPlaceFix]);
108             x2 = Integer.parseInt(splitBySpaces[4 + parameterPlaceFix]);
109             y2 = Integer.parseInt(splitBySpaces[5 + parameterPlaceFix]);
110 
111             if (method == 2 || method == 4) {
112                 startX = Integer.parseInt(splitBySpaces[6 + parameterPlaceFix]);
113                 startY = Integer.parseInt(splitBySpaces[7 + parameterPlaceFix]);
114                 if (isBetween(startX, x1, x2) == false || isBetween(startY, y1, y2) == false) {
115                     myLogger.insert("The start point needs to be in the area: " + commandLine);
116                     return "3rror"; //aloituspiste ei alueella
117                 }
118                 startPoint.x = startX;
119                 startPoint.y = startY;
120             }
121 
122         } catch (Exception e) {
123             myLogger.insert("Syntax error6: " + commandLine);
124             return "3rror"; //epäkelpoja numeroita tai parametrejä puuttuu
125         }
126 
127         if (isBetween(x1, 0, screenWidth) == false
128                 || isBetween(y1, 0, screenHeight) == false
129                 || isBetween(x2, x1, screenWidth) == false
130                 || isBetween(y2, y1, screenHeight) == false) {
131             myLogger.insert("The points need to be placed within the correct pixel range: " + commandLine + "\nThe correct range for x is from 0 to " + screenWidth + " and for y from 0 to " + screenHeight);
132             return "3rror"; //pisteet eivät ole näytön alueella
133         }
134 
135         if (x1 == x2 || y1 == y2) {
136 
137             myLogger.insert("Rectangle width and height must be > 0: " + commandLine);
138             return "3rror";
139         }
140 
141         Rectangle area = new Rectangle();
142         area.setRect(x1, y1, x2 - x1, y2 - y1);
143         Point colorCoodrinates = new Point(-1, -1);
144 
145         switch (method) {
146             case 1:
147                 colorCoodrinates = colorOperation.findColor(colorOperation.hexToRGB(splitBySpaces[2]), area);
148                 break;
149             case 2:
150                 colorCoodrinates = colorOperation.findColorStartingFromPoint(colorOperation.hexToRGB(splitBySpaces[2]), startPoint, area);
151                 break;
152             case 3:
153                 colorCoodrinates = colorOperation.findColorFromPalette(colorPalette, area);
154                 break;
155             case 4:
156                 colorCoodrinates = colorOperation.findColorFromPaletteStartingFromPoint(colorPalette, startPoint, area);
157                 break;
158         }
159 
160         return colorCoodrinates.x + " " + colorCoodrinates.y;
161     }
162 
163     /**
164      * This is used to check if a given integer value is inbetween the given
165      * parameters.
166      *
167      * @param value The value which is compared to the parameters
168      * @param low Parameter for the lowest possible number (inclusive)
169      * @param high Parameter for the highest possible number (inclusive)
170      * @return true if the value is between the given parameters. False if the
171      * number is not between the given parameters.
172      */
173     private boolean isBetween(int value, int low, int high) {
174         if (value <= high && value >= low) {
175             return true;
176         } else {
177             return false;
178         }
179     }
180 
181     /**
182      * Checks if the given parameter is a valid number in hex format. For
183      * example #FFFFFF and #000000 are valid.
184      *
185      * @param hex The String to check
186      * @return True if String is a hex number, false if not.
187      */
188     public boolean isValidHex(String hex) {
189         for (int i = 1; i < hex.length(); i++) {
190             if (Character.digit(hex.charAt(i), 16) == -1) {
191                 return false;
192             }
193         }
194         return true;
195     }
