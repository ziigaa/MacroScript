6   package macroscript.logic;
7   
8   import macroscript.gui.frmLogger;
9   import java.util.HashMap;
10  
11  /**
12   * For handling if-sentences within the script.
13   *
14   * @author Joonas
15   */
16  public class IfHandler {
17  
18      HashMap<String, Integer> variablesAndValues;
19      private final frmLogger myLogger;
20      private final FindColorOperator findColorOperation;
21  
22      /**
23       *      * The constructor.
24       *
25       * @param variablesAndValues A HashMap containing all the variables and
26       * their values.
27       * @param findColorOperation The color finding tool.
28       * @param myLogger The error logger.
29       */
30      public IfHandler(HashMap<String, Integer> variablesAndValues, FindColorOperator findColorOperation, frmLogger myLogger) {
31          this.variablesAndValues = new HashMap<>();
32          this.variablesAndValues = variablesAndValues;
33          this.myLogger = myLogger;
34          this.findColorOperation = findColorOperation;
35      }
36  
37      /**
38       * Interprets an if-sentence input from the script by splitting it and
39       * processing the remaining data.
40       *
41       * @param ifSentence The sentence to interpret
42       * @return If the requirements of the if/sentence are met, then the command
43       * after the condition is returned. Otherwise an empty string is returned.
44       */
45      public String handleIf(String ifSentence) {
46  
47          String toReturn = "";
48  
49          try {
50              String ifSplittedBySpaces[] = ifSentence.split(" ");
51              String splitByEqual[] = ifSplittedBySpaces[1].split("=");
52  
53              int givenValue = 0;
54  
55              try {
56                  givenValue = Integer.parseInt(splitByEqual[1]);
57              } catch (NumberFormatException e) {
58                  if (variablesAndValues.containsKey(splitByEqual[1])) {
59                      givenValue = variablesAndValues.get(splitByEqual[1]);
60                  }
61              }
62  
63              if (variablesAndValues.containsKey(splitByEqual[0])) {
64                  if (variablesAndValues.get(splitByEqual[0]) == givenValue) {
65                      for (int i = 2; i < ifSplittedBySpaces.length; i++) {
66                          toReturn = toReturn + ifSplittedBySpaces[i] + " ";
67                      }
68                      toReturn = toReturn.substring(0, toReturn.length() - 1);
69                  }
70              } else {
71                  myLogger.insert("Variable not found in: " + ifSentence);
72                  toReturn = "3rror";
73                  //variable not found
74              }
75          } catch (ArrayIndexOutOfBoundsException e) {
76              //ei "="-merkkiä
77              String ifSplittedBySpaces[] = ifSentence.split(" ");
78              if (ifSplittedBySpaces[1].contains("findColor")) {
79                  //System.out.println("löyty findColor");
80                  String colorPoint = findColorOperation.handleFindColorAsParameter(ifSentence);
81                  if (!colorPoint.contains("-1")) {
82                      try {
83                          //System.out.println("ifS: " + ifSentence);
84  
85                          int beginCommand = 7;
86  
87                          if (ifSplittedBySpaces[1].contains("FromPalette")) {
88                              beginCommand--;
89                          }
90  
91                          if (ifSentence.contains("StartingFromPoint")) {
92                              beginCommand += 2;
93                          }
94  
95                          for (int i = beginCommand; i < ifSplittedBySpaces.length; i++) {
96                              toReturn = toReturn + ifSplittedBySpaces[i] + " ";
97                          }
98                          toReturn = toReturn.substring(0, toReturn.length() - 1);
99                      } catch (Exception ex) {
100                         myLogger.insert("Invalid format in: " + ifSentence + "\nIf-sentence needs to have a condition");
101                         //if-lauseella ei ollut ehtoa
102                     }
103                 } else {
104                     myLogger.insert("Color not found: " + ifSentence);
105                     //väriä ei löytynyt
106                     toReturn = "3rror";
107                 }
108             } else {
109                 myLogger.insert("Invalid format in: " + ifSentence);
110             }
111         } catch (NullPointerException e) {
112             //ei ollu parametrii
113             myLogger.insert("Invalid parameter format in: " + ifSentence);
114             toReturn = "3rror";
115         }
116         return toReturn;
117     }
118 }
