6   package operators;
7   
8   import java.awt.AWTException;
9   import java.awt.Robot;
10  import static java.awt.event.KeyEvent.*;
11  import java.util.Random;
12  import java.util.logging.Level;
13  import java.util.logging.Logger;
14  
15  /**
16   * The keyboard functions are executed here.
17   *
18   * @author Joonas
19   */
20  public class KeyboardOperator {
21  
22      private Robot bot;
23      private boolean isHuman = false;
24      private boolean keyFunctionDown = false;
25      private boolean keyFunctionUp = false;
26  
27      /**
28       * The constructor.
29       *
30       * @throws AWTException If the Robot fails.
31       */
32      public KeyboardOperator() throws AWTException {
33          this.bot = new Robot();
34          this.isHuman = false;
35      }
36  
37      /**
38       * Sends a key down message that corresponds a key from the keyboard.
39       *
40       * @param theKey The key to press
41       */
42      public void keyDown(char theKey) {
43          this.keyFunctionDown = true;
44          typeChar(theKey, this.isHuman, this.keyFunctionDown, this.keyFunctionUp);
45      }
46  
47      /**
48       * Sends a key up message that corresponds a key from the keyboard.
49       *
50       * @param theKey The key to press
51       */
52      public void keyUp(char theKey) {
53          keyFunctionUp = true;
54          typeChar(theKey, this.isHuman, this.keyFunctionDown, this.keyFunctionUp);
55      }
56  
57      /**
58       * Sends an enter key press.
59       *
60       * @param isHuman If true, adds a random Thread.Sleep() before and after the
61       * key down event.
62       */
63      public void pressEnter(boolean isHuman) {
64          if (isHuman == true) {
65              doRndSleep();
66          }
67  
68          bot.keyPress(VK_ENTER);
69          if (isHuman == true) {
70              doRndSleep();
71          }
72          bot.keyRelease(VK_ENTER);
73      }
74  
75      /**
76       * Sends messages that replicate the function of the keyboard and uses this
77       * to type the input.
78       *
79       * @param characters The input to type
80       * @param isHuman If true, adds a random Thread.Sleep() before and after
81       * every key down event, including shift if a special character needs it or
82       * if the character in question is upper case.
83       */
84      public void type(CharSequence characters, boolean isHuman) {
85          int length = characters.length();
86          this.isHuman = isHuman;
87          for (int i = 0; i < length; i++) {
88              char character = characters.charAt(i);
89              typeChar(character, isHuman, keyFunctionDown, keyFunctionUp);
90          }
91      }
92  
93      /**
94       * Finds the correct keycode for the specified character and either presses,
95       * presses and releases or releases the key adding random Thread.sleep()
96       * commands on the flow according to given parameters.
97       *
98       * @param charToType The character to solve the keycode for. Also the
99       * character to undergo the specified action as mentioned above.
100      * @param isHuman True if random Thread.sleep() times are wanted, false if
101      * not.
102      * @param onlyDown If the only action wanted is a keyDown function.
103      * @param onlyUp If the only action wanted is a keyUp function.
104      */
105     private void typeChar(char charToType, boolean isHuman, boolean onlyDown, boolean onlyUp) {
106         String alphas = "abcdefghijklmnopqrstuvwxyz";
107         String numbers = "0123456789";
108         String specialCharSet1 = "!\"#Â¤%&/()=";
109         String specialCharSet2 = ",.-";
110         int charInt = 0;
111         boolean shiftNeeded = false;
112 
113         if (alphas.contains(Character.toString(charToType))) {
114             charInt = alphas.indexOf(charToType) + VK_A;
115         } else if (alphas.toUpperCase().contains(Character.toString(charToType))) {
116             charInt = alphas.toUpperCase().indexOf(charToType) + VK_A;
117             shiftNeeded = true;
118         } else if (numbers.contains(Character.toString(charToType))) {
119             charInt = numbers.indexOf(charToType) + VK_0;
120         } else if (charToType == ' ') {
121             charInt = VK_SPACE;
122         } else if (specialCharSet1.contains(Character.toString(charToType))) {
123             shiftNeeded = true;
124 
125             charInt = VK_0 + 1 + specialCharSet1.indexOf(charToType);
126 
127             if (charToType == '=') {
128                 charInt = VK_0;
129             }
130         } else if (specialCharSet2.contains(Character.toString(charToType))) {
131             if (charToType == '.') {
132                 charInt = VK_PERIOD;
133             } else if (charToType == ',') {
134                 charInt = VK_COMMA;
135             } else if (charToType == '-') {
136                 charInt = VK_MINUS;
137             }
138         }
139         if (isHuman == true) {
140             doRndSleep();
141         }
142         if (shiftNeeded == true) {
143             bot.keyPress(VK_SHIFT);
144             if (isHuman == true) {
145                 doRndSleep();
146             }
147         }
148         if (onlyDown == true) {
149             onlyDown = false;
150             bot.keyPress(charInt);
151             return;
152         }
153         if (onlyUp == true) {
154             onlyUp = false;
155             bot.keyRelease(charInt);
156             return;
157         }
158 
159         bot.keyPress(charInt);
160         if (isHuman == true) {
161             doRndSleep();
162         }
163 
164         bot.keyRelease(charInt);
165 
166         if (shiftNeeded == true) {
167             if (isHuman == true) {
168                 doRndSleep();
169             }
170             bot.keyRelease(VK_SHIFT);
171             shiftNeeded = false;
172         }
173     }
174 
175     /**
176      * Adds a Thread.sleep() with a randomized time between 0 and 300 ms.
177      */
178     private void doRndSleep() {
179         Random rnd = new Random();
180         long sInterval = (long) rnd.nextInt(300);
181 
182         try {
183             Thread.sleep(sInterval);
184         } catch (InterruptedException ex) {
185             Logger.getLogger(MouseOperator.class.getName()).log(Level.SEVERE, null, ex);
186         }
187     }
188 }
